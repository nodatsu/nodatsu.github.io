メインタブ

int scene;

void setup() {
  size(400, 400);
  scene = 0;
}

void draw() {
  switch (scene) {
  case 0:
    opening();
    break;
  case 1:
    playing();
    break;
  case 2:
    result();
    break;
  default:
    exit();
  }
}

--------------------------------------------------
openingタブ
--------------------------------------------------
void opening() {
  background(0);
  fill(255);

  text("OPENING", 200, 200);
}

--------------------------------------------------

--------------------------------------------------
void playing() {
  background(0);
  fill(255);

  text("PLAYING", 200, 200);
}
