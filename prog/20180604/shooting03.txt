--------------------------------------------------
メインタブ
--------------------------------------------------
int scene;

void setup() {
  size(400, 400);
  scene = 0;
  rectMode(CENTER);
}

void draw() {
  switch (scene) {
  case 0:
    opening();
    break;
  case 1:
    playing();
    break;
  case 2:
    result();
    break;
  default:
    exit();
  }
}

--------------------------------------------------
openingタブ
--------------------------------------------------
void opening() {
  background(0);
  fill(255);

  text("OPENING", 200, 200);
  
  if (mousePressed) {
    scene = 1;
  }
}

--------------------------------------------------
playingタブ
--------------------------------------------------
int eN = 3;

void playing() {
  background(0);
  fill(255);

  text("enemies = " + eN, 300, 20);

  rect(200, 50, 30, 30);
  
  rect(200, 350, 30, 30);
  
  ellipse(200, 250, 15, 15);

  if (keyPressed) {
    scene = 2;
  }
}

--------------------------------------------------
resultタブ
--------------------------------------------------
void result() {
  background(0);
  fill(255);

  text("RESULT", 200, 200);

  if (mousePressed) {
    scene = 3;
  }
}
